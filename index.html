function onEditSyntheseAuto(e) { if (!e || !e.source || !e.range) return; var sheet = e.range.getSheet(); var nom = sheet.getName().toUpperCase(); var col = e.range.getColumn(); var row = e.range.getRow(); // Onglets a ignorer var ignorer = [ "SYNTHESE","IMPORT","NOTES","BULLETIN","ANNONCE", "ABSENCE","MESSAGES","PHOTOS","RECUS","JOURNAL", "NOUVEAUX","ALERTES","WHATSAPP","EMARG","CARTES", "TABLEAU","STATISTIQUE","DASHBOARD","HISTORIQUE", "SANS_CONTACT","CSV_SMS","MATRICULES","ELEVES_MANQUANTS" ]; for (var i = 0; i < ignorer.length; i++) { if (nom.indexOf(ignorer[i]) >= 0) return; } if (row < 4 || col < 1 || col > 10) return; // ============================================================ // VERROU ANTI-SATURATION — CORRECTION CRITIQUE // ============================================================ var lock = LockService.getScriptLock(); if (!lock.tryLock(3000)) { Logger.log("onEditSyntheseAuto: verrou non obtenu, abandon"); return; } try { var rowData = sheet.getRange(row, 1, 1, 10).getValues()[0]; if (!rowData[0] || isNaN(Number(rowData[0]))) return; var mat = String(rowData[1] || "").trim(); var nom2 = String(rowData[2] || "").trim(); var du = parseFloat(rowData[3]) || 0; var vs = parseFloat(rowData[4]) || 0; var sl = parseFloat(rowData[5]) || 0; var t1 = String(rowData[6] || "").trim(); var t2 = String(rowData[7] || "").trim(); var t3 = String(rowData[8] || "").trim(); var stat = String(rowData[9] || "").trim(); if (!mat && !nom2) return; if (!stat) { stat = sl <= 0 ? "SOLDE" : vs <= 0 ? "IMPAYE" : "PARTIEL"; } // ============================================================ // 1. METTRE A JOUR SYNTHESE // ============================================================ var ss = e.source; var wsSynth = ss.getSheetByName("SYNTHESE"); if (wsSynth && wsSynth.getLastRow() >= 2) { var synthData = wsSynth.getRange(2, 2, wsSynth.getLastRow() - 1, 1).getValues(); var ligneFound = -1; for (var r = 0; r < synthData.length; r++) { if (String(synthData[r][0]).trim().toUpperCase() === mat.toUpperCase()) { ligneFound = r + 2; break; } } var newData = [sheet.getName(), mat, nom2, du, vs, sl, t1, t2, t3, stat]; if (ligneFound > 0) { wsSynth.getRange(ligneFound, 1, 1, 10).setValues([newData]); } else { wsSynth.getRange(Math.max(wsSynth.getLastRow() + 1, 2), 1, 1, 10).setValues([newData]); } } // ============================================================ // 2. TRI ALPHABETIQUE — UNIQUEMENT SI NOM MODIFIE (col 3) // CORRECTION CRITIQUE : evite la boucle infinie // ============================================================ if (col === 3) { var lastRowSheet = sheet.getLastRow(); if (lastRowSheet < 5) return; var nbEleves = lastRowSheet - 3; var data = sheet.getRange(4, 1, nbEleves, 10).getValues(); var eleves = data.filter(function(r) { return r[2] && String(r[2]).trim() !== ""; }); if (eleves.length === 0) return; eleves.sort(function(a, b) { var na = String(a[2] || "").trim().toUpperCase(); var nb2 = String(b[2] || "").trim().toUpperCase(); return na.localeCompare(nb2, "fr"); }); eleves.forEach(function(r, i) { r[0] = i + 1; }); sheet.getRange(4, 1, nbEleves, 10).clearContent(); sheet.getRange(4, 1, eleves.length, 10).setValues(eleves); } } catch(err) { Logger.log("onEditSyntheseAuto: " + err.message); } finally { lock.releaseLock(); } }
